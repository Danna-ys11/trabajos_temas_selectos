For each role, find the average number of years employed by employees in that roleSELECT role,    AVG(years_employed) AS "Average No.Years employed"FROM employeesGROUP BY role;3. Find the total number of employee years worked in each buildingSELECT building,    SUM(years_employed)AS "total of employee years worked"FROM employeesGROUP BY building;
